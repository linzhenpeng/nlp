#!/usr/bin/env/python
# -*- coding: utf-8 -*- 
import os
import re
import string
import numpy as np
from pyltp import *
from zhon.hanzi import punctuation
 
 
class Preprocess(object):
    '''
    中文BLP预处理类
    '''
 
    # 处理数据的正则表达式
    DIGIT_RE = re.compile(r'\d+')  # 匹配数字
    LETTER_RE = re.compile(r'[a-zA-Z]+')  # 匹配任意大小写字母
    SPECIAL_SYMBOL_RE = re.compile('[^A-Z^a-z^0-9^\u4e00-\u9fa5]')
    # ^ 上尖号是^脱字符，表示不匹配集合的字符，
    #  \w           : 匹配数字、字母、和下划线，效果同[0-9a-zA-Z_]
    # \s            : 匹配任何的空白符(空格、换行、回车、换业、制表)，效果和[ \f\n\r\t]
    # \u4e00-\u9fa5是所有汉字的unicode编码范围
    NAMED_ENTITY = re.compile(r'[SBIE]+')
    STOPS = ['。', '.', '?', '？', '!', '！']  # 中英文句末字符
 
    # 句子所限制的最小和最大长度
    SENTENCE_MIN_LEN = 5
    SENTENCE_MAX_LEN = 50
 
    def __init__(self, ltp_model_dir):
        self._cws_model_path = os.path.join(ltp_model_dir, 'cws.model')
        self._pos_model_path = os.path.join(ltp_model_dir, 'pos.model')
        self._ner_model_path = os.path.join(ltp_model_dir, 'ner.model')
        self._par_model_path = os.path.join(ltp_model_dir, 'parser.model')
        self._srl_model_path = os.path.join(ltp_model_dir, 'pisrl_win.model')
 
    '''
    def __init__(self, ltp_model_dir):
        self._cws_model_path = os.path.join(ltp_model_dir, 'cws.model')
        self._pos_model_path = os.path.join(ltp_model_dir, 'pos.model')
        self._ner_model_path = os.path.join(ltp_model_dir, 'ner.model')
        self._srl_model_path = os.path.join(ltp_model_dir, 'srl.model')
        self._par_model_path = os.path.join(ltp_model_dir, 'parser.model')
    '''
 
    @staticmethod
    def read_text_file(text_file):
        """读取文本文件,并返回由每行文本作为元素组成的list."""
        with open(text_file, 'r', encoding='utf-8') as file:
            lines = [line.strip() for line in file]
        return lines
 
    @staticmethod
    def write_text_file(text_list,target_file ):
        """
        文本写入列表
        Argrgs：
            text_list:列表，每个元素是一条文本
            target_file:字符串,写入的路径
        """
        with open(target_file, 'w', encoding='utf-8') as writer:
            for text in text_list:
                writer.write(text+'\n')
 
    @staticmethod
    def merge_files(filedir, target_file):
        """
        合并一个文件夹中的文本文件，注意：需要合并的每个文件的结尾要有换行符
        :param filedir: 需要合并文件的文件夹
        :param target_file: 合并后写入目标的文件
        :return: None
        """
        filenames = os.listdir(filedir)
        with open(target_file, 'a', encoding='utf-8') as f:
            for filename in filenames:
                filepath = os.path.join(filedir, filename)
                f.writelines(open(filepath, encoding='utf-8').readlines())
 
    @staticmethod
    def partition_dataset(dataset,ratio):
        """
        将一个大数集按比例切分为训练集、验证集、测试集
        :param dataset: 列表，原始数据集
        :param ratio: 三元组，训练集、验证集、测试集切分比例，每个元素为0-1之间的小说
        :return: train, val, test, 表示训练集、验证集、测试集的三个列表
        """
        data_len = len(dataset)
        train_len = int(np.floor(dataset * ratio[0])) # np.floor是取不大于传入数的最大整数
        val_len = int(np.ceil(data_len * ratio[1])) # np.ceil是取不小于传入数的最小整数
        test_len = data_len - train_len - val_len
        return dataset[:train_len], dataset[train_len:val_len], dataset[val_len:test_len]
 
    @staticmethod
    def is_equal(sent1, sent2):
        """
        判断两个句子是否完全相同
        :param sent1:
        :param sent2:
        :return:
        """
        return sent1 == sent2
 
    @staticmethod
    def del_blank_lines(sentences):
        """
        删除句子列表中的空行(\n、\t、\r、\f)，返回没有空行的句子列表
        :param sentences: 字符串列表
        :return:
        """
        return [s for s in sentences if s.split()]
 
    @ staticmethod
    def proprecess_chinese(sentence, letters=False, digital=False):
        """
        该函数可以根据具体要求具体添加，这里默认只保留汉字，可以通过传参
        是否保留英文字母和数字等，也可以在此基础上继续添加特定的功能
        :param sentence:
        :param letters: 默认为False，不保留
        :param digital: 默认为False，不保留
        :return: 处理好的字符串
        """
        if letters & digital:
            pro_str = re.compile('[^A-Z^a-z^0-9^\u4e00-\u9fa5]')
        elif ~letters & digital:
            pro_str = re.compile('[^0-9^\u4e00-\u9fa5]')
        elif letters & ~digital:
            pro_str = re.compile('[^A-Z^a-z^\u4e00-\u9fa5]')
        else:
            pro_str = re.compile('[^\u4e00-\u9fa5]')
        return pro_str.sub('', sentence)
 
    @staticmethod
    def del_stopwords(seg_sents, stopwords):
        """
            Args:
                seg_sents: 嵌套列表，分好词的句子（列表）的列表
                stopwords: 停用词列表
            Returns: 去除了停用词的句子的列表
        """
        data = [[word for word in sent if word not in stopwords] for sent in seg_sents]
        return data
 
 
    @classmethod
    def is_length_valid(cls, sentence):
        """
        限制句子的长度，判断是否合法
        :param sentence:
        :return:
        """
        return cls.SENTENCE_MIN_LEN <= len(sentence) <= cls.SENTENCE_MAX_LEN
 
    @classmethod
    def is_simple_sentence(cls, sentence):
        """
        判断是否是简单句
        简单句在这里定义为句子中只有一个句末终止符的句子，这样的句子含义比较明确
        :param sentence:
        :return:
        """
        counter = 0
        for word in sentence:
            if word in cls.STOPS:
                counter += 1
                if counter > 1:
                    return False
 
        return True
 
    # 下面这个两个功能都可以在proprecess_chinese进行实现，这里不细讲了
    @classmethod
    def del_special_symbol(cls, sentence):
        """删除句子中的乱码和一些特殊符号。"""
        return cls.SPECIAL_SYMBOL_RE.sub('', sentence)
 
    @classmethod
    def del_english_word(cls, sentence):
        """删除句子中的英文字符"""
        return cls.LETTER_RE.sub('', sentence)
 
    # 下面就是简单句子处理了，像命名体识别、分词处理、词性标注等
    # 这里使用的包不一样，后面我单独使用其他的方式进行实现
 
    @classmethod
    def get_ne_index(cls, ne_sent):
        """
        获取命名实体在句子中的位置
        :param ne_sent:  命名实体标记构成的列表
        :return:
        """
        return [idw for idw, word in enumerate(ne_sent) if cls.NAMED_ENTITY.match(word)]
 
    def seg_sentences(self, sentences):
        """
        对输入的字符串列表进行分词处理，返回分词后的字符串列表
        :param sentences:
        :return:
        """
        segmentor = Segmentor()
        segmentor.load(self._cws_model_path)
        seg_sents = [list(segmentor.segment(sent)) for sent in sentences]
        segmentor.release()
        return seg_sents
 
    def postag_sentences(self, seg_sents):
         """
        对分词后的句子列表进行词性标注，返回标注的词性列表
        :param seg_sents: 分好词的语句列表，每个语句也是一个列表
        :return:
        """
         postagger = Postagger()
         postagger.load(self._pos_model_path)
         pos_sents  = [list(postagger.postag(sent)) for sent in seg_sents]
         postagger.release()
         return pos_sents
 
    def rec_named_entity(self, seg_sents, pos_sents):
        """
        命名体识别
        :param seg_sents: 分词好的语句列表，每个语句也是一个列表
        :param pos_sents: 词性标注好的词性列表，每个语句的词性也是一个列表
        :return: 命名实体识别完的语句列表，每个语句的命名实体识别结果也是一个列表
        """
        recognizer = NamedEntityRecognizer()
        recognizer.load(self._ner_model_path)
        ne_sents = [list(recognizer.recognize(seg_sents[i], pos_sents[i])) for i in range(len(seg_sents))]
        recognizer.release()
        return ne_sents
 
    def parse_dependency(self, seg_sents, pos_sents):
         """
        依存句法分析
        :param seg_sents: 分好词的语句列表，每个语句也是列表
        :param pos_sents: 词性标注好的词性列表，每个语句的词性也是一个列表
        :return:
        arc_objs: pyltp.VectorOfParseResult对象，依存句法分析结果对象的列表。
        arc_sents: 依存句法分析完的语句列表,每个语句的依存句法分析结果也是一个列表
        """
         parser = Parser()
         parser.load(self._par_model_path)
         arc_objs = [parser.parse(seg_sents[i],pos_sents[i]) for i in range(len(seg_sents))]
         arc_sents = [[(a.head, a.relation) for a in arc] for arc in arc_objs]
         parser.release()
         return arc_objs, arc_sents
 
    def label_sementic_role(self, seg_sents, pos_sents, arc_sents):
        """
        语意角色标注
        :param seg_sents:  分好词的语句列表，每个语句也是一个列表
        :param pos_sents:  词性标注好的词性列表，每个语句的词性也是一个列表
        :param arc_sents:  依存句法分析结果列表，每个语句的依存句法分析结果也是一个列表
        :return: 语义角色标注完的语句列表，每个语句的语义角色标注结果也是一个列表
        """
        labeler = SementicRoleLabeller()
        labeler.load(self._srl_model_path)
 
        roles = [labeler.label(seg_sents[i], pos_sents[i], arc_sents[i]) for i in range(len(seg_sents))]
        _ret = []
        for role in roles:
            _role = []
            for r in role:
                _role.extend([(r.index, arg.name,arg.range.start, arg.range.end) for arg in r.arguments])
            _ret.append(_role)
        labeler.release()
        return _ret
